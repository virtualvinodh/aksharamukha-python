𑤁𑤧𑥀𑤛𑥂𑤱𑤧𑤛𑤾𑤟𑤰𑤟𑤳𑤬𑤾𑤤𑥂𑤳𑤛𑤱𑤬𑤴𑤛𑥂𑤤𑤽

(𑤍) 𑤬𑤻𑤬𑤾𑤌𑥂𑤳𑤛𑤠𑤰𑤖𑤭𑤽

𑤣𑤰𑤧𑤛𑤲𑤥𑤣𑤰𑤫𑤰𑤥𑤰𑤤𑤽-𑤁𑤧𑥀𑤛𑥂𑤱𑤧𑤛𑤾𑤟𑤰𑤟𑤳𑤬𑤾𑤤𑥂𑤳𑤛𑤱𑤬𑤴𑤛𑥂𑤤𑤽

𑤣𑤸𑤖𑤣𑤰𑤫𑤰𑤥𑤰𑤤𑤽-𑤡𑤎𑤾𑤬𑤽-𑤠𑤰-𑤝𑤌𑤸𑤟𑤽-𑤤𑤾𑤑𑤾𑤒𑤸𑤎𑤽-𑤎𑤾𑤬𑤳𑤤𑤽-𑤓𑤵𑤬𑤽-𑤬𑤳-𑤘𑥂𑤟𑤽-𑤠𑤰-𑤤𑤾𑤝𑤸

𑤟𑤤𑤭𑤽 𑤬𑥁𑤩𑤢𑤳𑤝𑤾𑤞𑤢𑤸𑤞𑤱𑤬𑤛𑤾𑤛𑤾𑤩𑤵𑤣𑥀𑤭𑤽

(𑤢𑤳𑤝𑤾𑤞𑤰𑤟𑤳𑤬𑤾𑤤𑥂𑤳𑤛𑤱𑤭𑤽)

𑤂𑤛𑥀𑤠𑤱 𑤢𑤳𑤝𑤾𑤞𑤸 𑤣𑤎𑤩𑤰𑤻𑤬𑤾𑤛𑤜𑤰𑤎𑤛𑤸𑥁𑤭𑤟𑤽 𑤬𑤤𑥀𑤌𑤾𑤬𑤻𑤢𑤳𑤝𑤾𑤞𑤸 𑤩𑤱𑤝𑥀𑤰𑤑𑤧𑤚𑤬𑤤𑤾𑤠𑤟𑤾𑤟𑤭𑤽 𑤬𑤳𑤎𑤛𑤸 𑤨𑤸𑤌𑤩𑤱𑤝𑤟𑤳𑤛𑤾𑤛𑤧𑤭𑤽 𑤠𑤳𑤧𑤳𑤫𑤝𑤤𑥀𑤬𑤰𑤧𑤜𑤱𑤭𑤽 𑤪𑤰𑤬𑤾𑤛𑤰 𑤝𑤵𑤩𑤤𑤟𑤳𑤫𑥀𑤰𑤚𑤰𑤻 𑤢𑤳𑤝𑤾𑤞𑤸 𑤣𑤎𑤩𑤰𑤟𑤱𑤛𑤱𑥆

𑤟𑤱𑤫𑥀𑤟𑤾𑤝𑤭𑤽 𑤬 𑤛𑤜𑤰𑤎𑤛𑤭𑤽 𑤠𑤳𑤚𑥀𑤰𑤟𑤰𑤤𑤽, 𑤀𑤩𑤱𑤠𑥂𑤚𑤰𑤪𑤭𑤽 𑤌𑤳𑤪𑤨𑤤𑤴𑤨𑤰𑤟𑤰𑤤𑤽, 𑤀𑤨𑤐𑤾𑤌𑥂𑤳𑤛𑤭𑤽 𑤌𑤾𑤫𑤰𑤟𑤾𑤛𑥀𑤰, 𑤁𑤨𑤥𑤭𑤽 𑤠𑤳𑤚𑥀𑤟𑤱𑤞𑤰𑤟𑤰𑤟𑤰𑤤𑤽, 𑤑𑤱𑤛𑥂𑤱𑤛𑤸𑤟𑤳𑤩𑥀𑤕𑤾𑤓𑤟𑤷𑤭𑤽 𑤌𑤳𑤬𑤳𑤤𑤱𑤛𑤸 𑤨𑤌𑤾𑤫𑤚𑤷𑤭𑤽, 𑤠𑥂𑤛𑤱𑤧𑤴𑤠𑤸 𑤎𑤸𑤑𑤧𑤵𑤚, 𑤀𑤠𑥂𑤛𑤱𑤌𑤴𑤨𑤸 𑤝𑥁𑤪𑤟𑤵𑤟, 𑤀𑤣𑤱𑤧𑤛𑤱𑤭𑤽 𑤪𑥂𑤝𑤾𑤞𑤰(𑤪𑤳𑤝𑤾𑤞𑤰)𑤞𑤱𑤤𑤳𑤌𑤾𑤛𑤰𑤟𑤰𑤤𑤽, 𑤀𑤟𑤣𑤱𑤣𑤴𑤛𑤭𑤽 𑤠𑥂𑤓𑤾𑤕𑤥𑤰, 𑤀𑤟𑤩𑤤𑥁𑤝𑤟𑤲𑤥𑤸 𑤢𑤨𑤷𑤭𑤽, 𑤪𑤰𑤬𑤾𑤛𑤰 𑤬𑥁𑤩𑤬𑤛𑤾𑤛𑤾𑤩𑤰𑤟𑤰𑤤𑤽, 𑤠𑤱𑤛𑤰 𑤢𑤸𑤞𑤱𑤬𑤛𑤾𑤛𑤾𑤩𑤰𑤟𑤰𑤤𑤽, 𑤧𑤰𑤓𑤰 𑤁𑤧𑥀𑤠𑤳𑤝𑤾𑤎𑤨𑤰𑤟𑤰𑤤𑤽, 𑤬𑤰𑥁𑤜𑤩𑤰𑤭𑤭𑤽 𑤟𑤱𑥁𑤩𑤰𑤚𑤟𑤎𑤧𑤬𑤤𑤾𑤠𑥂𑤬𑤾𑤜𑤱𑤛𑤰𑤟𑤰𑤤𑤽, 𑤀𑤠𑥂𑤤𑤵𑤥𑤸 𑤓𑤾𑤕𑤰𑤟𑤵𑤟, 𑤀𑤑𑤱𑤟𑤾𑤛𑥀𑤭𑤽 𑤠𑥂𑤛𑤱𑤣𑤰𑤟𑤵𑤟, 𑤩𑤱𑤪𑤳𑤝𑤾𑤞𑤭𑤽 𑤬𑤾𑤩𑤧𑤵𑤚, 𑤁𑤬𑤾𑤩𑤝𑤟𑤲𑤥𑤸 𑤏𑤸𑤫𑤵𑤚, 𑤀𑤬𑤵𑤑𑤟𑤌𑤸 𑤧𑤴𑤠𑤵𑤚, 𑤀𑤠𑥂𑤛𑤱𑤬𑤤𑤭𑤽 𑤌𑤰𑤥𑤵𑤟, 𑤀𑤨𑤱𑤠𑤾𑤛𑤭𑤽 𑤌𑤰𑤤𑤷𑤭𑤽, 𑤀𑤟𑤳𑤠𑤨𑤱𑤠𑤾𑤛𑤸 𑤧𑤴𑤠𑤷𑤭𑤽 𑤀𑤬𑤻𑤬𑥂𑤳𑤫𑤾𑤖 𑤁𑤧𑤴𑤠𑥀𑤷𑤭𑤽, 𑤩𑤱𑤠𑥂𑤤𑤳𑤌𑤾𑤛𑤭𑤽 𑤬𑤾𑤌𑤟𑤾𑤞𑤵𑤣𑥀𑤭𑤽, 𑤩𑤱𑤬𑤤𑤾𑤠𑥂𑤥𑤳𑤌𑤾𑤛𑤸 𑤞𑤰𑤛𑤳𑤣𑤱𑤭𑤽, 𑤬𑤻𑤩𑥂𑤳𑤛 𑤁𑤥𑤛𑤟𑤷𑤭𑤽, 𑤠𑥂𑤑𑤾𑤒𑤱𑤟𑤾𑤟𑤸 𑤎𑥂𑤟𑤾𑤜𑤷𑤭𑤽, 𑤩𑤱𑤤𑤳𑤌𑤾𑤛𑤭𑤽 𑤠𑤧𑤱𑤝𑤰𑤏𑤾𑤟𑤷𑤭𑤽, 𑤠𑤧𑤱𑤤𑤳𑤌𑤾𑤛𑤬𑤾𑤛𑥂𑤳𑤫𑤾𑤚𑤥𑤰, 𑤉𑤏𑤰𑤝𑤳𑤛𑤾𑤛𑤲𑥁𑤚𑤭𑤽 𑤠𑤧𑤱𑤠𑤴𑥁𑤚𑤸 𑤓𑤾𑤕𑤰𑤟𑤵𑤟, 𑤠𑥂𑤛𑤱𑤫𑤾𑤖𑤱𑤛𑤸𑤛𑤲𑤛𑤰𑤟𑤰𑤎𑤛𑤠𑥂𑤛𑥀𑤳𑤛𑤾𑤠𑤟𑤾𑤟𑤰𑤟𑤰𑤻 𑤢𑤳𑤝𑤾𑤞𑤰𑤟𑤰𑤻 𑤣𑤎𑤩𑤛𑤰𑤻 𑤓𑤾𑤕𑤰𑤟𑤵, 𑤀𑤠𑥂𑤛𑤱𑤫𑤾𑤖𑤱𑤛𑤸 𑤟𑤱𑥁𑤩𑤰𑤚𑤵, 𑤬𑤾𑤜𑤱𑤛𑤸 𑤣𑤴𑤛𑤌𑤸𑤖𑥀𑤰𑤤𑤽, 𑤬𑤾𑤜𑤱𑤛𑤭𑤽 𑤬𑥁𑤩𑤬𑤛𑤾𑤛𑤾𑤩𑤰𑤨𑤸𑤌𑤟𑤲𑤥𑤰𑤥𑤰𑤻 𑤣𑤴𑤤𑤄, 𑤬𑥁𑤩 𑤂𑤤𑤵 𑤛𑤜𑤰𑤎𑤛𑤰𑤟𑤰𑤻 𑤩𑤱𑤪𑤵𑤫𑤛𑤭𑤽 𑤬𑤤𑥀𑤎𑤽 𑤎𑤳𑤚𑤰𑤭𑤽𑥆

(𑤞𑥁𑤤𑤰𑤟𑤳𑤬𑤾𑤤𑥂𑤳𑤛𑤱𑤭𑤽)

𑤬𑤝𑤾𑤞𑥁𑤤𑤬𑤾𑤛𑤳 𑤁𑤝𑤄 𑤌𑤨𑥀𑤰𑤚𑤭𑤽, 𑤤𑤞𑥀𑤵 𑤌𑤨𑥀𑤰𑤚𑤭𑤽, 𑤠𑤧𑥀𑤩𑤬𑤰𑤟𑤵 𑤌𑤨𑥀𑤰𑤚𑤭𑤽, 𑤬𑤾𑤩𑥁𑤜𑤭𑤽 𑤬𑤳𑤩𑥀𑤕𑤾𑤓𑤟𑤭𑤽, 𑤌𑤵𑤩𑤨𑤭𑤽, 𑤠𑤧𑤱𑤠𑤴𑥁𑤚𑤭𑤽, 𑤠𑤧𑤱𑤪𑤳𑤝𑤾𑤞𑤭𑤽, 𑤠𑤧𑥀𑤩𑤝𑤰𑤛𑤭𑤽, 𑤬𑤾𑤩𑤰𑤍𑥀𑤰𑤛𑤭𑤽 𑤣𑤎𑤩𑤛𑤸 𑤞𑥁𑤤𑤭𑤽, 𑤬𑤰𑤟𑤾𑤝𑥂𑤳𑤫𑤾𑤖𑤱𑤌𑤭𑤽, 𑤟𑤱𑥁𑤓𑤾𑤩𑤧𑤭𑤽, 𑤁𑤌𑤰𑤨𑤱𑤌𑤭𑤽, 𑤀𑤄𑤠𑤟𑤰𑤥𑤱𑤌𑤭𑤽, 𑤆𑤵𑤭𑤱𑤠𑤪𑥀𑤱𑤌𑤭𑤽, 𑤠𑥂𑤛𑥀𑤰𑤛𑤾𑤤𑤩𑤵𑤝𑤟𑤲𑤥𑤸 𑤩𑤱𑤓𑤾𑤕𑤷𑤭𑤽, 𑤬𑤾𑤩𑤰𑤍𑥀𑤰𑤛𑤸 𑤣𑤎𑤩𑤛𑤸 𑤞𑥁𑤤𑤩𑤱𑤟𑤥𑤭𑤽 𑤬𑤳𑤠𑥂𑤩𑤵𑤝𑤱𑤛𑤭𑤽 𑤟𑤷𑤧𑥀𑤰𑤚𑤱𑤌𑤭𑤽, 𑤬𑤻𑤢𑤸𑤞𑤱𑤎𑤰𑤤𑤲, 𑤀𑤣𑤱𑤟𑤾𑤟𑤭𑤽 𑤬𑤻𑤬𑤾𑤛𑤴𑤠𑤭𑤽, 𑤬𑤠𑥂𑤛𑤱𑤪𑤧𑤚𑤭𑤽, 𑤒𑤱𑤟𑤾𑤟𑤠𑤾𑤨𑤸𑤛𑤱𑤌𑤭𑤽𑥆

(𑤬𑤻𑤏𑤰𑤟𑤳𑤬𑤾𑤤𑥂𑤳𑤛𑤱𑤭𑤽)

𑤬𑤳𑤠𑥂𑤛𑤱𑤠𑤟𑤾𑤟𑤸 𑤣𑤎𑤩𑤛 𑤁𑤧𑥀𑤬𑤻𑤏𑤭𑤽, 𑤟𑥀𑤰𑤥𑤠𑥂𑤛𑤱𑤠𑤟𑤾𑤟𑤭𑤽, 𑤧𑤳𑤓𑤳𑤠𑥂𑤛𑤱𑤠𑤟𑤾𑤟𑤭𑤽, 𑤬𑤰𑤤𑤲𑤑𑤲𑤠𑥂𑤛𑤱𑤠𑤟𑤾𑤟𑤭𑤽, 𑤀𑤕𑤾𑤓𑤨𑤲𑤌𑤧𑤚𑤲𑤥𑤭𑤽, 𑤬𑤰𑤤𑤲𑤑𑤲𑤌𑤧𑤚𑤲𑤥𑤭𑤽, 𑤠𑤳𑤚𑥀𑤪𑥂𑤲𑤌𑤾𑤫𑤵𑤛𑥂𑤭𑤽, 𑤤𑤭𑤰𑤝𑤌𑤾𑤫𑤱𑤚𑤰𑤠𑤧𑤱𑤪𑤸𑤞𑤌𑤭𑤽, 𑤠𑥂𑤰𑤭𑤩𑤟𑤲𑤥𑤭𑤽, 𑤁𑤭𑤩𑤟𑤲𑤥𑤭𑤽𑥆
