𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤭𑥂𑤳𑤝𑤥𑤬𑤴𑤛𑥂𑤤𑤽𑥆

[𑤬𑤻𑤌𑤾𑤫𑤱𑤠𑤾𑤛𑤤𑤰𑤛𑥂𑤳𑤌𑤰]

𑥆 𑤟𑤤𑤭𑤽 𑤬𑥁𑤩𑤓𑤾𑤕𑤰𑤥 𑥆

𑥆 𑤉𑤼 𑤟𑤤𑤸 𑤣𑤎𑤩𑤛𑥀𑤷 𑤁𑤧𑥀𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤥𑤷 𑥆

𑤁𑤧𑥀𑤰𑤩𑤨𑤸𑤌𑤱𑤛𑤵𑤪𑤾𑤩𑤧𑤢𑤸𑤞𑤱𑤬𑤛𑤾𑤛𑤾𑤩𑤸 𑤎𑤤𑤾𑤣𑤲𑤧𑤰𑤥𑤰𑤻 𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤥𑤰𑤻 𑤑𑤧𑥀𑤰𑤻 𑤑𑤧𑤤𑤰𑤚𑤸 𑤩𑥀𑤩𑤨𑤸𑤌𑤥𑤛𑤱 𑤬𑤾𑤤𑥆 𑤠𑤕𑤾𑤑 𑤬𑤾𑤌𑤟𑤾𑤞𑤰𑤭𑤽, 𑤛𑤰𑤻𑤪𑤾𑤑 𑤬𑤾𑤩𑤣𑤰𑤩𑤪𑤴𑤟𑥀𑤰𑤟𑤽 𑤠𑤪𑥀𑤛𑤱 𑤬𑤾𑤤𑥆

𑤂𑤭 𑤪𑤰𑤧𑤱𑤠𑤳𑤛𑥂 𑤧𑤴𑤠𑤻 𑤪𑤴𑤟𑥀𑤛𑤰, 𑤪𑤴𑤟𑥀𑤛𑤷𑤩 𑤧𑤴𑤠𑤤𑤽𑥆 𑤧𑤴𑤠𑤰𑤟𑤾𑤟 𑤠𑥂𑤳𑤜𑤌𑤽 𑤪𑤴𑤟𑥀𑤛𑤰, 𑤪𑤴𑤟𑥀𑤛𑤰𑤥𑤰 𑤟 𑤠𑥂𑤳𑤜𑤎𑤽 𑤧𑤴𑤠𑤤𑤽𑥆 𑤥𑤝𑥂𑤴𑤠𑤻 𑤬𑤰 𑤪𑤴𑤟𑥀𑤛𑤰, 𑤥𑤰 𑤪𑤴𑤟𑥀𑤛𑤰 𑤛𑤝𑥂𑤴𑤠𑤤𑤽𑥆

𑤆𑤩𑤤𑤵𑤩 𑤩𑤵𑤝𑤟𑤰𑤬𑤻𑤓𑤾𑤕𑤰𑤬𑤻𑤬𑤾𑤌𑤰𑤧𑤩𑤱𑤓𑤾𑤕𑤰𑤟𑤰𑤟𑤱𑥆

𑤂𑤭𑤻 𑤪𑤰𑤧𑤱𑤠𑤳𑤛𑥂 𑤬𑥁𑤩𑤞𑥁𑤤𑤰𑤭𑤽 𑤪𑤴𑤟𑥀𑤛𑤰𑤨𑤌𑤾𑤫𑤚𑤰 𑤀𑤟𑤳𑤛𑤾𑤠𑤟𑤾𑤟𑤰 𑤀𑤟𑤱𑤧𑤳𑤝𑤾𑤞𑤰 𑤀𑤤𑤨𑤰 𑤟 𑤩𑤱𑤤𑤨𑤰 𑤟𑤸𑤟𑤰 𑤟 𑤠𑤧𑤱𑤠𑤴𑥁𑤚𑤰𑤭𑤽𑥆 𑤛𑤬𑤾𑤤𑤰𑤑𑤾𑤒𑤰𑤧𑤱𑤠𑤳𑤛𑥂 𑤪𑤴𑤟𑥀𑤛𑤰𑤥𑤰𑤻 𑤟 𑤧𑤴𑤠𑤤𑤽, 𑤟 𑤩𑤵𑤝𑤟𑤰, 𑤟 𑤬𑤻𑤓𑤾𑤕𑤰, 𑤟 𑤬𑤻𑤬𑤾𑤌𑤰𑤧𑤰𑤭𑤽, 𑤟 𑤩𑤱𑤓𑤾𑤕𑤰𑤟𑤰𑤟𑤱𑥆 𑤟 𑤑𑤌𑤾𑤫𑤳𑤭𑤽𑤪𑥂𑤸𑤛𑥂𑤏𑥂𑤰𑤚𑤓𑤱𑤭𑤾𑤩𑤰𑤌𑤰𑤥𑤤𑤟𑤰𑤻𑤬𑤱, 𑤟 𑤧𑤴𑤠𑤪𑤢𑤾𑤝𑤎𑤟𑤾𑤞𑤧𑤬𑤬𑤾𑤠𑥂𑤫𑤾𑤖𑤩𑥀𑤞𑥁𑤤𑤰𑤭𑤽𑥆 𑤟 𑤑𑤌𑤾𑤫𑤳𑥁𑤞𑤰𑤛𑤳𑤧𑥀𑤰𑤩𑤟𑤾𑤟 𑤤𑤟𑤸𑤞𑤰𑤛𑤳𑤭𑤽𑥆

𑤟 𑤩𑤱𑤝𑥀𑤰 𑤟𑤰𑤩𑤱𑤝𑥀𑤰 𑤟 𑤩𑤱𑤝𑥀𑤰𑤌𑤾𑤫𑤥𑤸 𑤟𑤰𑤩𑤱𑤝𑥀𑤰𑤌𑤾𑤫𑤥𑤸 𑤥𑤰𑤩𑤟𑤾𑤟 𑤓𑤧𑤰𑤤𑤧𑤚𑤻 𑤟 𑤓𑤧𑤰𑤤𑤧𑤚𑤌𑤾𑤫𑤥𑤸 𑤟 𑤝𑤳𑤭𑤽𑤍𑤬𑤤𑤳𑤝𑤥𑤟𑤱𑤧𑤸𑤞𑤤𑤰𑥁𑤎𑤰 𑤟 𑤓𑤾𑤕𑤰𑤟𑤻 𑤟 𑤠𑥂𑤰𑤠𑤾𑤛𑤱𑤛𑤾𑤩𑤤𑤽𑥆

𑤢𑤸𑤞𑤱𑤬𑤛𑤾𑤛𑤾𑤩𑤬𑥀 𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤤𑤰𑤪𑥂𑤱𑤛𑥀 𑤩𑤱𑤭𑤧𑤛𑤱 𑤑𑤱𑤛𑤾𑤛𑤰𑤩𑤧𑤚𑤭𑤽𑥆 𑤑𑤱𑤛𑤾𑤛𑤰𑤩𑤧𑤚𑤟𑤰𑤬𑤾𑤛𑤱𑤛𑤾𑤩𑤰𑤝𑤛𑥂𑤬𑤾𑤛𑤸 𑤩𑤱𑤠𑤧𑥀𑤰𑤬𑤰𑤛𑤱𑤌𑥂𑤰𑤟𑤾𑤛𑤸 𑤟𑤱𑤫𑤾𑤖𑤟𑤱𑥁𑤩𑤰𑤚𑤭𑤽𑥆 𑤛𑥂𑥀𑤞𑤾𑤩𑤩𑥀𑤩𑤬𑤾𑤜𑤱𑤛𑤰𑤭𑤽 𑤬𑥁𑤩𑤢𑤳𑤝𑤾𑤞𑤰𑤭𑤽 𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤤𑤰𑤪𑥂𑤱𑤛𑥀 𑤀𑤟𑤳𑤛𑤾𑤛𑤧𑤰𑤻 𑤬𑤤𑥀𑤌𑤾𑤬𑤻𑤢𑤸𑤞𑤱𑤤𑤣𑤱𑤬𑤻𑤢𑤳𑤝𑤾𑤞𑤰𑤭𑤽𑥆

𑤛𑤬𑤾𑤤𑤰𑤓𑤾𑤓𑤾𑤕𑤰𑤛𑤩𑥀𑤭𑤽 𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤤𑤭𑤰𑤤𑤟𑤾𑤛𑥂𑤸 𑤤𑤭𑤰𑤩𑤱𑤝𑥀𑤰𑤤𑤟𑤾𑤛𑥂𑤸𑤟𑤳𑤛𑤾𑤛𑤧𑤤𑤟𑤾𑤛𑥂𑤸𑤬𑤤𑤬𑤤𑤤𑤟𑤾𑤛𑥂𑤭𑤽 𑤬𑥁𑤩𑤝𑤳𑤭𑤽𑤍𑤠𑥂𑤪𑤤𑤟𑤭𑤽 𑤬𑤛𑥀𑤤𑤤𑤱𑤜𑥀𑤛𑤾𑤩𑤰𑤛𑤽 𑤠𑥂𑤓𑤾𑤕𑤰𑤠𑤰𑤧𑤤𑤱𑤛𑤰𑤥𑤰𑤤𑤳𑤌𑤾𑤛𑤸 𑤤𑤟𑤾𑤛𑥂𑤭𑤽𑥆
