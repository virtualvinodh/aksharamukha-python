𑼭𑼭𑼶𑼡𑼮𑼶𑼱𑽂𑼡𑼬𑼃

𑼥𑼪𑼃 𑼱𑼂𑼮𑼨𑼸𑼣𑽂𑼤𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼾𑼩𑽂𑼫𑼃𑽃

𑽄 𑼐𑼀 𑼥𑼪𑼾𑼴 𑼣𑼯𑼣𑼶𑼔𑼥𑼥𑽂𑼡𑼴𑼦𑼂𑼫𑼥𑽂𑼡𑼭𑼾𑼴𑼒𑼤𑼴𑼡𑼸𑼦𑽂𑼬𑼡𑼶𑼰𑽂𑼝𑼶𑼡𑼱𑼂𑼮𑼨𑼸𑼣𑽂𑼤𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼴𑼂𑼫𑼯𑽂𑼬𑼴𑼮𑼒𑼦𑽂𑼬𑼡𑽂𑼫𑼾𑼒𑼨𑼸𑼣𑽂𑼤𑼾𑼩𑽂𑼫𑼾𑼴𑼡𑼷𑼡𑼴𑼥𑼴𑼔𑼡𑼦𑽂𑼬𑼡𑽂𑼫𑼸𑼡𑽂𑼦𑼥𑽂𑼥𑼾𑼩𑽂𑼫𑼃𑽄

𑽑𑽐 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼱𑼁𑼣𑼂𑼯𑼥𑼦𑼬𑼶𑼮𑼂𑼡𑼾𑼴 𑼣𑼯𑼪𑼃𑽃

𑼆𑼡𑼶 𑼲𑼶 𑼩𑼶𑼒𑽂𑼰𑼮𑼃 𑼱𑼁𑼮𑼺𑼣𑽂𑼤𑼃 𑼒𑼸𑼪𑼴𑼬𑼃𑽃 𑼡𑼣𑼴 𑼪𑼴𑼖𑽂𑼔𑼭𑽂𑼫𑼯𑼡𑼱𑼲𑼱𑽂𑼬𑼿𑼃 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼪𑼸𑼦𑼥𑼷𑼫𑼡𑼾 𑼱𑽂𑼪 𑼣𑼯𑼩𑼶𑼂𑼣𑼴𑼬𑼒𑼱𑼲𑼱𑽂𑼬𑼿𑼃 𑼦𑼬𑼶𑼮𑼺𑼡𑼃 𑼦𑼸𑼬𑼱𑽂𑼒𑼺𑼡𑼃, 𑼣𑼯𑼩𑼶𑼯𑽂𑼗 𑼬𑼢𑼱𑼲𑼱𑽂𑼬𑼿𑼃 𑼓𑼴𑼣𑼥𑼷𑼫𑼩𑼾𑼴𑼙𑼥𑼷𑼫𑼱𑽂𑼮𑼴𑼣𑼥𑼷𑼫𑼦𑼬𑼶𑼦𑼹𑼂𑼠𑼿𑼂𑼲𑼶𑼬𑼠𑽂𑼫𑼱𑼸𑼮𑼂𑼠𑼦𑼬𑼶𑼦𑼹𑼂𑼠𑼿𑼯𑽂𑼗𑽃 𑼫𑼾𑼥 𑼒𑼦𑼶𑼭𑼮𑼱𑽂𑼡𑼸𑼥𑼶 𑼪𑼲𑼴𑼥𑼔𑼬𑼾 𑼮𑼷𑼢𑼶𑼗𑼡𑽂𑼮𑼬𑼬𑼢𑽂𑼫𑼴𑼥𑽂𑼡𑼬𑼴𑼦𑼠𑼪𑼸𑼓𑼾𑼰𑽂𑼮𑼩𑽂𑼫𑼮𑼒𑼷𑼂𑼫𑼡𑼾 𑼱𑽂𑼪 𑼄𑼩𑼶𑼮𑼶𑼯𑽂𑼬𑼴𑼪𑽂𑼫𑼥𑽂𑼡𑼾𑽃 𑼄𑼰𑽂𑼜𑼵𑼩𑼶𑼯𑽂𑼗 𑼡𑼹𑼂𑼫𑼯𑼡𑼱𑼲𑼱𑽂𑼬𑼿𑼃 𑼦𑽂𑼬𑼕𑼸𑼰𑽂𑼫𑼪𑼴𑼠𑼿𑼂𑼪𑼲𑼡𑼴 𑼗 𑼦𑼸𑼰𑽂𑼦𑼮𑼂𑼰𑼾𑼠𑼴𑼩𑼶𑼦𑽂𑼬𑼮𑼂𑼰𑼡𑼴 𑼮𑼶𑼡𑼂𑼣𑼶𑼥𑼶𑼂𑼫𑼹𑼲𑼡𑼾𑼴𑼬𑼠𑼔𑼮𑼴𑼒𑽂𑼰𑼲𑼂𑼪𑽂𑼫𑼒𑼹𑼜𑼵𑼔𑼴𑼬𑼦𑽂𑼬𑼴𑼱𑼴𑼣𑼡𑼭𑼾𑼰𑼸 𑼒𑼥𑽂𑼫𑼴𑼯𑼡𑼱𑼲𑼱𑽂𑼬𑼴𑼠𑼶 𑼱𑼂𑼮𑼴𑼭𑼁𑼒𑼴𑼬𑼩𑼹𑼰𑼶𑼡𑼴𑼃 𑼱𑽂𑼢𑼶𑼡𑼴 𑼄𑼩𑼹𑼮𑼥𑽁𑽃 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼁 𑼦𑽂𑼬𑼾𑼒𑽂𑼰𑼪𑼴𑼠𑼴𑼃 𑼒𑼸𑼱𑼸𑼪𑼴𑼥𑼶 𑼗 𑼒𑽂𑼰𑼶𑼦𑼥𑽂𑼡𑼶 𑼱𑽂𑼪𑽃 𑼄𑼰𑽂𑼜𑼿𑼴 𑼗 𑼪𑼬𑼸𑼡𑽂𑼒𑼥𑽂𑼫𑼴𑼱𑼲𑼱𑽂𑼬𑼴𑼠𑼶 𑼮𑼶𑼔𑼭𑼶𑼡𑼴𑼭𑼁𑼒𑼴𑼬𑼴𑼩𑼬𑼠𑼴𑼭𑼁𑼒𑼺𑼡𑼴𑼥𑼶 𑼬𑼡𑽂𑼥𑼩𑼣𑽂𑼬𑼁𑼒𑼬𑼾𑼠 𑼔𑼺𑼲𑼷𑼡𑼴𑼥𑼶 𑼪𑼴𑼂𑼔𑼁 𑼯𑼾𑼴𑼤𑼫𑼥𑽂𑼡𑽂𑼫𑼾𑼴 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼱𑽂𑼫 𑼦𑼸𑼬𑼡𑼾𑼴 𑼔𑼗𑽂𑼘𑼥𑽂𑼡𑼶 𑼱𑽂𑼪𑽃 𑼣𑼾𑼮𑼥𑼴𑼔𑼫𑼒𑽂𑼰𑼔𑼥𑽂𑼤𑼂𑼮𑼴𑼱𑼸𑼬𑼔𑼬𑼸𑼞𑼒𑼶𑼥𑽂𑼥𑼬𑼪𑼲𑼾𑼴𑼬𑼔𑼴𑼯𑽂𑼗𑼴𑼂𑼤𑼒𑼴𑼫𑼶𑼒𑼴 𑼔𑼔𑼥𑼡𑼭𑼴𑼡𑽂𑼦𑼸𑼰𑽂𑼦𑼦𑼜𑽂𑼜𑼣𑼴𑼪𑼴𑼥𑽂𑼫𑼩𑼶𑼦𑽂𑼬𑼭𑼪𑽂𑼨𑼫𑼥𑽂𑼡𑼶 𑼱𑽂𑼪𑽃 𑼱𑼂𑼮𑼾 𑼗 𑼯𑼴𑼒𑽂𑼫𑼔𑼠𑼴𑼃 𑼯𑼸𑼣𑽂𑼤𑼾𑼴𑼣𑼥𑼁 𑼬𑼴𑼙𑼴𑼥𑼁 𑼦𑼸𑼬𑼱𑽂𑼒𑼺𑼡𑽂𑼫 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼱𑽂𑼫 𑼦𑼸𑼬𑼡𑼾𑼴 𑼔𑼗𑽂𑼘𑼥𑽂𑼡𑼶 𑼱𑽂𑼪𑽃 𑼄𑼥𑼾𑼥𑼿𑼮𑼁𑼬𑼹𑼦𑼾𑼠 𑼮𑽂𑼫𑼹𑼲𑼾𑼥 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼾𑼴 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼪𑼸𑼦𑼥𑼷𑼫𑼡𑼾 𑼱𑽂𑼪𑽄

𑼱𑼪𑼥𑼥𑽂𑼡𑼬𑼦𑽂𑼬𑼮𑼾𑼯𑼶𑼡𑼯𑽂𑼗 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼾𑼴 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼪𑽁𑽃 𑼄𑼢 𑼮𑼶𑼯𑽂𑼮𑼴𑼪𑼶𑼡𑽂𑼬𑼾𑼴 𑼥𑼴𑼪 𑼣𑼴𑼬𑼒𑼴𑼗𑼴𑼂𑼫𑼾𑼴 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼱𑽂𑼫 𑼯𑽂𑼬𑼶𑼫𑼁 𑼡𑼾𑼙𑼯𑽂𑼗𑼴𑼱𑼲𑼪𑼴𑼥𑼾𑼴 𑼤𑼬𑼠𑼶𑼡𑼭𑼾 𑼥𑼶𑼮𑼶𑼰𑽂𑼜𑼾𑼴𑼤𑼾𑼴𑼪𑼸𑼓𑼃 𑼦𑽂𑼬𑼦𑼡𑼡𑼶 𑼱𑽂𑼪𑽃 𑼡𑼁 𑼡𑼢𑼴 𑼦𑽂𑼬𑼦𑼡𑼶𑼡𑼁 𑼣𑼺𑼰𑽂𑼜𑽂𑼮𑼴 𑼯𑼸𑼩𑼴𑼖𑽂𑼔𑼾𑼴 𑼥𑼴𑼪 𑼡𑼸𑼰𑼶𑼡𑼒𑼴𑼫𑼶𑼒𑼾𑼴 𑼣𑼾𑼮𑼦𑼸𑼡𑽂𑼬𑼾𑼴 𑼣𑼒𑽂𑼰𑼶𑼠𑼾𑼥 𑼒𑼬𑼡𑼭𑼾𑼥 𑼦𑼬𑼶𑼔𑼺𑼲𑽂𑼫𑼾𑼴𑼡𑽂𑼢𑼴𑼦𑼫𑼡𑼶 𑼱𑽂𑼪𑽃 𑼈𑼡𑽂𑼢𑼴𑼦𑽂𑼫 𑼗 𑼔𑼔𑼥𑼡𑼭𑼱𑽂𑼢𑼾𑼴 𑼬𑼴𑼙𑼴𑼥𑼁 𑼯𑼸𑼣𑽂𑼤𑼾𑼴𑼣𑼥𑼁 𑼡𑼁 𑼗 𑼪𑼲𑼴𑼥𑽂𑼡𑼁 𑼙𑼥𑼒𑼴𑼫𑼁 𑼔𑼴𑼢𑼴𑼩𑼶𑼬𑼩𑽂𑼫𑼩𑼴𑼰𑼡𑽁—

𑼯𑼴𑼱𑽂𑼡𑽂𑼬𑼴𑼠𑼶 𑼫𑼴𑼥𑼶 𑼦𑽂𑼬𑼗𑼭𑼥𑽂𑼡𑼶 𑼪𑼥𑼸𑼰𑽂𑼫𑼭𑼾𑼴𑼒𑼾
𑼱𑼁𑼓𑽂𑼫𑼴 𑼭𑼶𑼦𑼶𑼯𑽂𑼗 𑼔𑼠𑼥𑼴𑼦𑼶 𑼗 𑼤𑼴𑼡𑼸𑼡𑼥𑽂𑼡𑽂𑼬𑼪𑽁𑽃
𑼫𑼾 𑼯𑼶𑼭𑽂𑼦𑼫𑼾𑼴𑼔 𑼦𑼺𑼢𑼸 𑼭𑼿𑼴𑼒𑼶𑼒 𑼄𑼦𑽂𑼬𑼪𑼾𑼫𑼴𑼃
𑼡𑼾𑼰𑽂𑼮𑼾𑼰𑼸 𑼯𑼶𑼒𑽂𑼰𑼶𑼡𑼸 𑼦𑼸𑼬𑼴 𑼨𑼲𑼸𑼒𑼭𑽂𑼦𑼒𑼾𑼴𑼜𑽂𑼫𑼃𑽄𑽑𑽄

𑼒𑼶𑼁 𑼡𑼹 𑼙𑼥𑼱𑽂𑼫 𑼄𑼥𑼸𑼮𑼂𑼡𑼥𑼡𑼴𑼁 𑼒𑼬𑼾𑼴𑼡𑼶
𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼪𑼴𑼔𑼡𑼸 𑼱𑼸𑼯𑼶𑼒𑽂𑼰𑼶𑼡𑼸 𑼯𑼶𑼰𑽂𑼫𑼠𑼴𑼂𑼢𑼪𑽁𑽃
𑼦𑼬𑼶𑼦𑼵𑼗𑼥𑼴𑼂𑼢 𑼨𑼲𑼸𑼣𑼴𑼬𑼒 𑼄𑼔𑽂𑼬𑼫𑼴𑼥𑼾
𑼄𑼥𑽂𑼫𑼴𑼁𑼯𑽂𑼗 𑼱𑼡𑽂𑼡𑽂𑼮𑼥𑼫𑼸𑼡𑼴𑼥𑼪𑼺𑼡𑼾 𑼮𑼶𑼥𑼾𑼡𑼸𑼪𑽁𑽄𑽒𑽄

𑼭𑼾𑼴𑼒𑼾𑼴𑼡𑽂𑼡𑼬𑼾𑼰𑼸 𑼗𑼡𑼸𑼱𑼡𑽂𑼫𑼦𑼢𑼾 𑼮𑼶𑼤𑼶𑼙𑽂𑼛𑼾𑼴
𑼲𑼾𑼡𑼸𑼦𑽂𑼬𑼡𑼷𑼡𑽂𑼫𑼒𑼸𑼯𑼭𑼾𑼴 𑼫𑼢 𑼱𑼁𑼩𑼮𑼥𑽂𑼡𑼶𑽃
𑼫𑼢 𑼗𑼴𑼥𑼶𑼬𑼾𑼴𑼤𑼒𑽂𑼰𑼫𑼸 𑼱𑼁𑼱𑽂𑼢𑼶𑼡𑼸 𑼯𑼷𑼡𑼶𑼩𑼴𑼮𑼃
𑼡𑼱𑽂𑼪𑼶𑼥𑽂𑼮𑼶𑼤𑼶𑼙𑽂𑼛 𑼒𑼶𑼪𑼢𑼾𑼴 𑼭𑼶𑼦𑼶𑼯𑼴𑼱𑽂𑼡𑽂𑼬𑼪𑼴𑼡𑽂𑼬𑼾𑽄𑽓𑽄

𑼥𑼾𑼡𑼱𑽂𑼫 𑼅𑼗𑼬𑼶𑼫 𑼈𑼡𑽂𑼡𑼬𑼶 𑼮𑼴 𑼡𑽂𑼬𑼶𑼭𑼾𑼴𑼒𑼾
𑼱𑼂𑼮𑼾𑼰𑼸 𑼣𑼾𑼮𑼪𑼥𑼸𑼙𑼾𑼰𑽂𑼮𑼫𑼪𑼾𑼮 𑼙𑼾𑼰𑽂𑼝𑼃𑽃
𑼥𑼴𑼪𑼴𑼦𑼶 𑼡𑼾𑼰 𑼭𑼶𑼦𑼶𑼥𑼴𑼁 𑼥 𑼲𑼶 𑼮𑼶𑼡𑽂𑼢 𑼫𑼹𑼫𑼁
𑼫𑼡𑽂𑼬𑼾𑼰𑼸 𑼯𑼶𑼒𑽂𑼰𑼶𑼡𑼸 𑼦𑼸𑼬𑼴 𑼨𑼲𑼸𑼒𑼭𑽂𑼦𑼒𑼾𑼴𑼜𑽂𑼫𑼃𑽄𑽔𑽄

𑼱𑼾𑼴 𑼗𑼶𑼡𑽂𑼡𑼤𑼴𑼬 𑼙𑼔𑼡𑼴𑼁 𑼮𑼶𑼮𑼶𑼤𑼴 𑼮𑼶𑼗𑼶𑼡𑽂𑼬𑼴
𑼎𑼒𑼒𑽂𑼰𑼠𑼾𑼥 𑼄𑼫𑼸 𑼙𑼴𑼥𑼡𑼶 𑼯𑼸𑼣𑽂𑼤𑼱𑼡𑽂𑼡𑽂𑼮𑼃𑽃
𑼄𑼣𑼺𑼯𑽂𑼫𑼬𑼹𑼦𑼬𑼲𑼶𑼡𑼱𑽂𑼫 𑼔𑼡𑼶𑼁 𑼗 𑼮𑼾𑼡𑽂𑼡𑼶
𑼒𑼶𑼁 𑼮𑼴 𑼦𑼸𑼥𑼾𑼴𑼢 𑼭𑼶𑼦𑼶𑼥𑼾𑼴𑼒𑽂𑼰𑼬𑼣𑼺𑼯𑽂𑼫𑼬𑼹𑼦𑼵𑼪𑽁𑽄𑽕𑽄

𑼆𑼡𑽂𑼫𑼸𑼒𑽂𑼡𑽂𑼮𑼴 𑼱 𑼣𑼾𑼮𑼦𑼸𑼡𑽂𑼬𑼾𑼴 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼁 𑼣𑼶𑼮𑽂𑼫𑼿𑼃 𑼒𑼸𑼱𑼸𑼪𑼿𑼬𑼩𑽂𑼫𑼂𑼗𑽂𑼫 𑼡𑼡𑽂𑼬𑼿𑼮𑼴𑼥𑽂𑼡𑼂𑼣𑼤𑼾𑽃 𑼡𑼡𑽂𑼬 𑼤𑼴𑼡𑽂𑼬𑽂𑼫𑼯𑽂𑼗 𑼗𑼾𑼜𑼷𑼮𑼂𑼔𑼴𑼯𑽂𑼗 𑼱𑽂𑼢𑼴𑼦𑼶𑼡𑼴 𑼄𑼩𑼹𑼮𑼥𑽁𑽃 𑼦𑼬𑼶𑼯𑼾𑼰𑼴𑼃 𑼯𑼴𑼒𑽂𑼫𑼴𑼃 𑼯𑼸𑼣𑽂𑼤𑼾𑼴𑼣𑼥𑼦𑽂𑼬𑼪𑼸𑼓𑼴𑼃 𑼦𑽂𑼬𑼒𑽂𑼬𑼴𑼪𑼥𑽂𑼡𑼃𑽄

𑼄𑼢 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮 𑼈𑼬𑼔𑼱𑼴𑼬𑼗𑼥𑽂𑼣𑼥𑼪𑼫𑼁 𑼭𑼶𑼦𑼶𑼧𑼭𑼒𑼪𑼴𑼣𑼴𑼫 𑼣𑼶𑼮𑽂𑼫𑼴𑼂𑼰𑼱𑼸𑼮𑼂𑼠𑼡𑼶𑼬𑼒𑼁 𑼱𑼪𑼥𑽂𑼡𑼴𑼥𑽂𑼪𑼠𑼶𑼬𑼡𑽂𑼥𑼦𑽂𑼬𑼡𑽂𑼫𑼸𑼦𑽂𑼡𑼁 𑼮𑼶𑼯𑽂𑼮𑼴𑼪𑼶𑼡𑽂𑼬𑼪𑼴𑼗𑼴𑼂𑼫𑼪𑼾𑼮𑼪𑼴𑼲-𑼒𑼡𑼪𑼴𑼁 𑼪𑼾 𑼩𑼾𑼴 𑼈𑼦𑼵𑼤𑽂𑼫𑼴𑼫 𑼭𑼶𑼦𑼶𑼁 𑼯𑼶𑼒𑽂𑼰𑼴𑼦𑼫𑼱𑼶𑽃 𑼨𑽂𑼬𑼴𑼲𑽂𑼪𑼷𑼓𑼬𑼾𑼴𑼰𑽂𑼜𑼷𑼦𑼸𑼰𑽂𑼒𑼬𑼱𑼴𑼬𑼶𑼁 𑼄𑼖𑽂𑼔𑼭𑼶𑼦𑼶𑼁 𑼮𑼖𑽂𑼔𑼭𑼶𑼦𑼶𑼁 𑼪𑼔𑼤𑼭𑼶𑼦𑼶𑼁 𑼪𑼖𑽂𑼔𑼭𑽂𑼫𑼭𑼶𑼦𑼶𑼁 𑼄𑼖𑽂𑼔𑼸𑼭𑼷𑼫𑼭𑼶𑼦𑼶𑼁 𑼯𑼒𑼴𑼬𑼶𑼭𑼶𑼦𑼶𑼁 𑼨𑽂𑼬𑼲𑽂𑼪𑼮𑼭𑼶𑼭𑼶𑼦𑼶𑼁 𑼦𑼵𑼬𑼸𑼰𑽂𑼫𑼭𑼶𑼦𑼶𑼁 𑼣𑽂𑼬𑼴𑼮𑼶𑼞𑼭𑼶𑼦𑼶𑼁 𑼒𑼶𑼬𑼴𑼡𑼭𑼶𑼦𑼶𑼁 𑼣𑼴𑼒𑽂𑼰𑼶𑼠𑽂𑼫𑼭𑼶𑼦𑼶𑼁 𑼈𑼔𑽂𑼬𑼭𑼶𑼦𑼶𑼁 𑼱𑼁𑼓𑽂𑼫𑼴𑼭𑼶𑼦𑼶𑼁 𑼄𑼥𑼸𑼭𑼾𑼴𑼪𑼭𑼶𑼦𑼶𑼁 𑼄𑼮𑼪𑼹𑼂𑼤𑼭𑼶𑼦𑼶𑼁 𑼣𑼬𑼣𑼭𑼶𑼦𑼶𑼁 𑼓𑼴𑼰𑽂𑼫𑼭𑼶𑼦𑼶𑼁 𑼗𑼷𑼥𑼭𑼶𑼦𑼶𑼁 𑼭𑼹𑼥𑼭𑼶𑼦𑼶𑼁 𑼲𑼹𑼠𑼭𑼶𑼦𑼶𑼁 𑼪𑼤𑽂𑼫𑼴𑼒𑽂𑼰𑼬𑼮𑼶𑼱𑽂𑼡𑼬𑼭𑼶𑼦𑼶𑼁 𑼦𑼸𑼰𑽂𑼦𑼭𑼶𑼦𑼶𑼁 𑼣𑼾𑼮𑼭𑼶𑼦𑼶𑼁 𑼥𑼴𑼔𑼭𑼶𑼦𑼶𑼁 𑼫𑼒𑽂𑼰𑼭𑼶𑼦𑼶𑼁 𑼔𑼥𑽂𑼤𑼂𑼮𑼭𑼶𑼦𑼶𑼁 𑼒𑼶𑼥𑽂𑼥𑼬𑼭𑼶𑼦𑼶𑼁 𑼪𑼲𑼾𑼴𑼬𑼔𑼭𑼶𑼦𑼶𑼁 𑼄𑼱𑼸𑼬𑼭𑼶𑼦𑼶𑼁 𑼔𑼬𑼸𑼞𑼭𑼶𑼦𑼶𑼁 𑼪𑼺𑼔𑼗𑼒𑽂𑼬𑼭𑼶𑼦𑼶𑼁 𑼮𑼴𑼫𑼱𑼬𑼸𑼡𑼭𑼶𑼦𑼶𑼁 𑼩𑼿𑼴𑼪𑼣𑼾𑼮𑼭𑼶𑼦𑼶𑼁 𑼄𑼥𑽂𑼡𑼬𑼷𑼒𑽂𑼰𑼣𑼾𑼮𑼭𑼶𑼦𑼶𑼁 𑼈𑼡𑽂𑼡𑼬𑼒𑼸𑼬𑼸𑼣𑽂𑼮𑼷𑼦𑼭𑼶𑼦𑼶𑼁 𑼄𑼦𑼬𑼔𑼾𑼴𑼞𑼴𑼥𑼷𑼭𑼶𑼦𑼶𑼁 𑼦𑼹𑼂𑼮𑼮𑼶𑼣𑼾𑼲𑼭𑼶𑼦𑼶𑼁 𑼈𑼡𑽂𑼒𑽂𑼰𑼾𑼦𑼭𑼶𑼦𑼶𑼁 𑼥𑼶𑼒𑽂𑼰𑼾𑼦𑼭𑼶𑼦𑼶𑼁 𑼮𑼶𑼒𑽂𑼰𑼾𑼦𑼭𑼶𑼦𑼶𑼁 𑼦𑽂𑼬𑼒𑽂𑼰𑼾𑼦𑼭𑼶𑼦𑼶𑼁 𑼱𑼴𑼔𑼬𑼭𑼶𑼦𑼶𑼁 𑼮𑼙𑽂𑼬𑼭𑼶𑼦𑼶𑼁 𑼭𑼾𑼓𑼦𑽂𑼬𑼡𑼶𑼭𑼾𑼓𑼭𑼶𑼦𑼶𑼁 𑼄𑼥𑼸𑼣𑽂𑼬𑼸𑼡𑼭𑼶𑼦𑼶𑼁 𑼯𑼴𑼱𑽂𑼡𑽂𑼬𑼴𑼮𑼂𑼡𑼴𑼁 𑼔𑼠𑼥𑼴𑼮𑼂𑼡𑼭𑼶𑼦𑼶𑼁 𑼈𑼡𑽂𑼒𑽂𑼰𑼾𑼦𑼵𑼮𑼂𑼡𑼭𑼶𑼦𑼶𑼁 𑼥𑼶𑼒𑽂𑼰𑼾𑼦𑼵𑼮𑼂𑼡𑼭𑼶𑼦𑼶𑼁 𑼦𑼵𑼣𑼭𑼶𑼓𑼶𑼡𑼭𑼶𑼦𑼶𑼁 𑼣𑽂𑼮𑼶𑼬𑼸𑼡𑽂𑼡𑼬𑼦𑼣𑼱𑼁𑼤𑼶𑼭𑼶𑼦𑼶𑼁 𑼫𑼴𑼮𑼣𑽂𑼣𑼯𑼾𑼴𑼡𑽂𑼡𑼬𑼦𑼣𑼱𑼁𑼤𑼶𑼭𑼶𑼦𑼶𑼁 𑼪𑼤𑽂𑼫𑼴𑼲𑼴𑼬𑼶𑼠𑼷𑼭𑼶𑼦𑼶𑼁 𑼱𑼂𑼮𑼬𑼸𑼡𑼱𑼁𑼔𑽂𑼬𑼲𑼠𑼷𑼭𑼶𑼦𑼶𑼁 𑼮𑼶𑼣𑽂𑼫𑼴𑼥𑼸𑼭𑼾𑼴𑼪𑼴𑼮𑼶𑼪𑼶𑼯𑽂𑼬𑼶𑼡𑼭𑼶𑼦𑼶𑼁 𑼊𑼰𑼶𑼡𑼦𑼱𑽂𑼡𑼦𑽂𑼡𑼴𑼁 𑼬𑼾𑼴𑼗𑼪𑼴𑼥𑼴𑼁 𑼤𑼬𑼠𑼷𑼦𑽂𑼬𑼾𑼒𑽂𑼰𑼶𑼠𑼷𑼭𑼶𑼦𑼶𑼁 𑼔𑼔𑼥𑼦𑽂𑼬𑼾𑼒𑽂𑼰𑼶𑼠𑼷𑼭𑼶𑼦𑼶𑼁 𑼱𑼂𑼮𑼿𑼴𑼰𑼤𑼶𑼥𑼶𑼰𑽂𑼫𑼥𑽂𑼣𑼴𑼁 𑼱𑼂𑼮𑼱𑼴𑼬𑼱𑼁𑼔𑽂𑼬𑼲𑼠𑼷𑼁 𑼱𑼂𑼮𑼩𑼹𑼡𑼬𑼸𑼡𑼔𑽂𑼬𑼲𑼠𑼷𑼪𑽁𑽃 𑼅𑼱𑼴𑼁 𑼩𑼾𑼴 𑼈𑼦𑼵𑼤𑽂𑼫𑼴𑼫 𑼗𑼡𑼸𑼰𑽂𑼰𑼰𑽂𑼜𑼷𑼭𑼶𑼦𑼷𑼥𑼴𑼁 𑼒𑼡𑼪𑼴𑼁 𑼡𑽂𑼮𑼁 𑼯𑼶𑼰𑽂𑼫𑼴𑼦𑼫𑼶𑼰𑽂𑼫𑼱𑼶?

𑼄𑼢 𑼮𑼶𑼯𑽂𑼮𑼴𑼪𑼶𑼡𑽂𑼬𑼾𑼴 𑼣𑼴𑼬𑼒𑼴𑼗𑼴𑼂𑼫𑼾𑼴 𑼮𑼶𑼱𑽂𑼪𑼶𑼡𑼃 𑼦𑽂𑼬𑼲𑼱𑼶𑼡𑼮𑼣𑼥𑼾𑼴 𑼥𑼶𑼲𑼡𑼪𑼴𑼥𑼪𑼣𑼣𑼂𑼦 𑼆𑼪𑼴𑼁 𑼔𑼴𑼢𑼴𑼪𑼩𑼴𑼰𑼡—

𑼅𑼯𑽂𑼗𑼂𑼫𑼁 𑼯𑼸𑼣𑽂𑼤𑼱𑼡𑽂𑼡𑽂𑼮𑼱𑽂𑼫 𑼭𑼾𑼴𑼒𑼾 𑼭𑼾𑼴𑼒𑼴𑼥𑼸𑼮𑼂𑼡𑼶𑼥𑼾𑼴𑽃
𑼯𑼶𑼒𑽂𑼰𑼶𑼡𑼃 𑼱𑼂𑼮𑼯𑼴𑼱𑽂𑼡𑽂𑼬𑼾𑼰𑼸 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼪𑼸𑼦𑼵𑼔𑼡𑼃𑽄𑽖𑽄

𑼫𑼾𑼰𑼴𑼪𑼲𑼁 𑼥𑼴𑼪𑼤𑼾𑼫𑼁 𑼭𑼶𑼦𑼷𑼥𑼴𑼁 𑼥 𑼦𑽂𑼬𑼙𑼴𑼥𑼪𑼶𑽃
𑼡𑼡𑽂𑼬𑼿𑼰 𑼯𑼶𑼒𑽂𑼰𑼶𑼡𑼃 𑼱𑼥𑽂𑼡𑼾𑼴 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼪𑼸𑼦𑼵𑼔𑼡𑼃𑽄𑽗𑽄

𑼮𑼒𑽂𑼡𑽂𑼬𑼁 𑼗𑼴𑼱𑽂𑼫 𑼥 𑼦𑼯𑽂𑼫𑼴𑼪𑼶 𑼪𑼹𑼂𑼤𑼴𑼥𑼁 𑼡𑼱𑽂𑼫 𑼥𑼿𑼮 𑼗𑽃
𑼯𑼶𑼰𑽂𑼫𑼫𑼶𑼰𑽂𑼫𑼾 𑼒𑼢𑼁 𑼲𑽂𑼫𑼾𑼥𑼁 𑼭𑼶𑼦𑼶𑼦𑽂𑼬𑼙𑽂𑼛𑼴𑼫 𑼦𑼵𑼬𑼔𑼪𑽁𑽄𑽘𑽄

𑼣𑼾𑼮𑼣𑼾𑼮𑼾𑼴 𑼲𑽂𑼫𑼡𑼶𑼣𑼾𑼮𑼃 𑼱𑼂𑼮𑼣𑼾𑼮𑼾𑼴𑼡𑽂𑼡𑼪𑼾𑼴 𑼮𑼶𑼩𑼸𑼃𑽃
𑼄𑼱𑼪𑼯𑽂𑼗 𑼮𑼶𑼯𑼶𑼰𑽂𑼜𑼯𑽂𑼗 𑼭𑼾𑼴𑼒𑼾𑼰𑽂𑼮𑼦𑽂𑼬𑼡𑼶𑼦𑼸𑼣𑽂𑼔𑼭𑼃𑽄𑽙𑽄

𑼄𑼱𑽂𑼫𑼿𑼮 𑼡𑽂𑼮𑼥𑼸𑼩𑼴𑼮𑼾𑼥 𑼦𑽂𑼬𑼙𑽂𑼛𑼾𑼴𑼦𑼵𑼫𑼾 𑼮𑼶𑼯𑼾𑼰𑼡𑼃𑽃
𑼯𑼶𑼒𑽂𑼰𑼶𑼡𑼁 𑼯𑼶𑼰𑽂𑼫𑼫𑼶𑼰𑽂𑼫𑼴𑼪𑼶 𑼱𑼂𑼮𑼭𑼾𑼴𑼒𑼦𑼬𑼴𑼫𑼠𑼪𑽁𑽄𑽑𑽐𑽄

𑼆𑼡𑼶 𑼲𑼶 𑼩𑼶𑼒𑽂𑼰𑼮𑼾𑼴 𑼣𑼯 𑼣𑼴𑼬𑼒𑼱𑼲𑼱𑽂𑼬𑼴𑼠𑼶 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼾𑼥 𑼱𑼴𑼂𑼤𑼁 𑼭𑼶𑼦𑼶𑼁 𑼯𑼶𑼰𑽂𑼫𑼥𑽂𑼡𑼾 𑼱𑽂𑼪𑽃 𑼡𑼡𑽂𑼬 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼴𑼤𑼶𑼱𑽂𑼢𑼴𑼥𑼾𑼥 𑼡𑼾𑼰𑼴𑼁 𑼣𑼴𑼬𑼒𑼴𑼠𑼴𑼁 𑼪𑼴𑼡𑼺𑼒𑼴𑼁 𑼮𑼴𑼗𑼫𑼡𑼴𑼁 𑼫𑼣𑼴 𑼄𑼒𑼴𑼬𑼁 𑼦𑼬𑼶𑼒𑼷𑼂𑼡𑼫𑼥𑽂𑼡𑼶 𑼱𑽂𑼪, 𑼡𑼣𑼴 𑼄𑼥𑼶𑼡𑽂𑼫𑼃 𑼱𑼂𑼮𑼱𑼁𑼱𑽂𑼒𑼴𑼬𑼯𑼨𑽂𑼣𑼾𑼴 𑼥𑼶𑼯𑽂𑼗𑼬𑼡𑼶 𑼱𑽂𑼪𑽃 𑼅𑼒𑼴𑼬𑼾 𑼦𑼬𑼶𑼒𑼷𑼂𑼡𑽂𑼫𑼪𑼴𑼥𑼾 𑼅𑼡𑽂𑼪𑼦𑼬𑼲𑼶𑼡𑼯𑼨𑽂𑼣𑼾𑼴 𑼥𑼶𑼯𑽂𑼗𑼬𑼡𑼶 𑼱𑽂𑼪𑽃 𑼆𑼒𑼴𑼬𑼾 𑼆𑼥𑽂𑼣𑽂𑼬𑼶𑼫𑼮𑼿𑼒𑼭𑽂𑼫𑼯𑼨𑽂𑼣𑼃𑽃 𑼇𑼒𑼴𑼬𑼾 𑼇𑼡𑼶𑼨𑼲𑼸𑼭𑼁 𑼙𑼔𑼣𑼶𑼡𑼶𑽃 𑼈𑼒𑼴𑼬𑼾 𑼈𑼦𑼣𑽂𑼬𑼮𑼨𑼲𑼸𑼭𑼁 𑼙𑼔𑼣𑼶𑼡𑼶𑽃 𑼉𑼒𑼴𑼬𑼾 𑼉𑼥𑼱𑼡𑽂𑼡𑽂𑼮𑼁 𑼙𑼔𑼣𑼶𑼡𑼶𑽃 𑼎𑼒𑼴𑼬𑼾 𑼎𑼰𑼠𑼴𑼱𑼪𑼸𑼡𑽂𑼢𑼴𑼥𑼣𑼾𑼴𑼰𑼯𑼨𑽂𑼣𑼃𑽃 𑼏𑼒𑼴𑼬𑼾 𑼏𑼂𑼫𑼴𑼦𑼢𑼃 𑼯𑽂𑼬𑼾𑼫𑼴𑼥𑼶𑼡𑼶𑽃 𑼐𑼒𑼴𑼬𑼾 𑼐𑼕𑼾𑼴𑼡𑽂𑼡𑼬𑼯𑼨𑽂𑼣𑼃𑽃 𑼐𑼴𑼒𑼴𑼬𑼾 𑼐𑼴𑼦𑼦𑼵𑼣𑼸𑼒𑼯𑼨𑽂𑼣𑼃𑽃 𑼄𑼁𑼒𑼴𑼬𑼾 𑼄𑼪𑼾𑼴𑼕𑼾𑼴𑼡𑽂𑼦𑼡𑽂𑼡𑼶𑼯𑼨𑽂𑼣𑼃𑽃 𑼄𑼃𑼒𑼴𑼬𑼾 𑼄𑼱𑽂𑼡𑼁𑼔𑼪𑼥𑼯𑼨𑽂𑼣𑼾𑼴 𑼥𑼶𑼯𑽂𑼗𑼬𑼡𑼶 𑼱𑽂𑼪𑽃 𑼒𑼒𑼴𑼬𑼾 𑼒𑼂𑼪𑼮𑼶𑼦𑼵𑼒𑼴𑼮𑼡𑼴𑼬𑼯𑼨𑽂𑼣𑼃𑽃 𑼓𑼒𑼴𑼬𑼾 𑼓𑼱𑼪𑼱𑼂𑼮𑼤𑼂𑼪𑼯𑼨𑽂𑼣𑼃𑽃 𑼔𑼒𑼴𑼬𑼾 𑼔𑼪𑽂𑼩𑼷𑼬𑼤𑼂𑼪𑼦𑽂𑼬𑼡𑼷𑼡𑽂𑼫𑼱𑼪𑼸𑼡𑽂𑼦𑼵𑼣𑼴𑼮𑼡𑼴𑼬𑼯𑼨𑽂𑼣𑼃𑽃 𑼕𑼒𑼴𑼬𑼾 𑼕𑼥𑼦𑼜𑼭𑼴𑼮𑼶𑼣𑽂𑼫𑼴𑼪𑼾𑼴𑼲𑼴𑼥𑽂𑼤𑼒𑼴𑼬𑼮𑼶𑼤𑼪𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼖𑼒𑼴𑼬𑼾𑼖𑽂𑼔𑼮𑼶𑼯𑼸𑼣𑽂𑼤𑼶𑼯𑼨𑽂𑼣𑼃𑽃 𑼗𑼒𑼴𑼬𑼾 𑼗𑼡𑼸𑼬𑼴𑼂𑼫𑼱𑼡𑽂𑼫𑼯𑼨𑽂𑼣𑼃𑽃 𑼘𑼒𑼴𑼬𑼾 𑼘𑼥𑽂𑼣𑼬𑼴𑼔𑼦𑽂𑼬𑼲𑼴𑼠𑼯𑼨𑽂𑼣𑼃𑽃 𑼙𑼒𑼴𑼬𑼾 𑼙𑼬𑼴𑼪𑼬𑼠𑼱𑼪𑼡𑼶𑼒𑽂𑼬𑼪𑼠𑼯𑼨𑽂𑼣𑼃𑽃 𑼚𑼒𑼴𑼬𑼾 𑼚𑼰𑼤𑽂𑼮𑼙𑼨𑼭𑼥𑼶𑼔𑽂𑼬𑼲𑼠𑼯𑼨𑽂𑼣𑼃𑽃 𑼛𑼒𑼴𑼬𑼾 𑼙𑽂𑼛𑼴𑼦𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼜𑼒𑼴𑼬𑼾 𑼦𑼜𑼾𑼴𑼦𑼗𑽂𑼘𑼾𑼣𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼝𑼒𑼴𑼬𑼾 𑼝𑼦𑼥𑼷𑼫𑼦𑽂𑼬𑼯𑽂𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼞𑼒𑼴𑼬𑼾 𑼞𑼪𑼬𑼪𑼴𑼬𑼥𑼶𑼔𑽂𑼬𑼲𑼠𑼯𑼨𑽂𑼣𑼃𑽃 𑼟𑼒𑼴𑼬𑼾 𑼪𑼷𑼟𑼮𑼶𑼰𑼫𑼴 𑼆𑼡𑼶𑽃 𑼠𑼒𑼴𑼬𑼾 𑼬𑼾𑼠𑼸𑼒𑽂𑼭𑼾𑼯𑼴 𑼆𑼡𑼶𑽃 𑼡𑼒𑼴𑼬𑼾 𑼡𑼢𑼡𑼴𑼱𑼁𑼩𑼾𑼣𑼯𑼨𑽂𑼣𑼃𑽃 𑼢𑼒𑼴𑼬𑼾 𑼢𑼴𑼪𑼨𑼭𑼮𑼾𑼔𑼮𑼿𑼯𑼴𑼬𑼣𑽂𑼫𑼯𑼨𑽂𑼣𑼃𑽃 𑼣𑼒𑼴𑼬𑼾 𑼣𑼴𑼥𑼣𑼪𑼱𑼁𑼫𑼪𑼱𑼿𑼴𑼬𑼩𑽂𑼫𑼯𑼨𑽂𑼣𑼃𑽃 𑼤𑼒𑼴𑼬𑼾 𑼤𑼥𑼪𑼴𑼂𑼫𑼴𑼠𑼴𑼁 𑼱𑼦𑽂𑼡𑼮𑼶𑼤𑼪𑼶𑼡𑼶𑽃 𑼥𑼒𑼴𑼬𑼾 𑼥𑼴𑼪𑼬𑼹𑼦𑼦𑼬𑼶𑼙𑽂𑼛𑼴𑼯𑼨𑽂𑼣𑼃𑽃 𑼦𑼒𑼴𑼬𑼾 𑼦𑼬𑼪𑼴𑼂𑼢𑼯𑼨𑽂𑼣𑼃𑽃 𑼧𑼒𑼴𑼬𑼾 𑼧𑼭𑼦𑽂𑼬𑼴𑼦𑽂𑼡𑼶𑼱𑼴𑼒𑽂𑼰𑼴𑼡𑽂𑼒𑽂𑼬𑼶𑼫𑼴𑼯𑼨𑽂𑼣𑼃𑽃 𑼨𑼒𑼴𑼬𑼾 𑼨𑼥𑽂𑼤𑼥𑼪𑼾𑼴𑼒𑽂𑼰𑼯𑼨𑽂𑼣𑼃𑽃 𑼩𑼒𑼴𑼬𑼾 𑼩𑼮𑼮𑼶𑼩𑼮𑼯𑼨𑽂𑼣𑼃𑽃 𑼪𑼒𑼴𑼬𑼾 𑼪𑼣𑼪𑼴𑼥𑼾𑼴𑼦𑼯𑼪𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼫𑼒𑼴𑼬𑼾 𑼫𑼢𑼴𑼮𑼣𑽂𑼤𑼂𑼪𑼦𑽂𑼬𑼡𑼶𑼮𑼾𑼤𑼯𑼨𑽂𑼣𑼃𑽃 𑼬𑼒𑼴𑼬𑼾 𑼬𑼡𑽂𑼫𑼬𑼡𑼶𑼦𑼬𑼪𑼴𑼂𑼢𑼬𑼡𑼶𑼯𑼨𑽂𑼣𑼃𑽃 𑼭𑼒𑼴𑼬𑼾 𑼭𑼡𑼴𑼘𑼾𑼣𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼮𑼒𑼴𑼬𑼾 𑼮𑼬𑼫𑼴𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼯𑼒𑼴𑼬𑼾 𑼯𑼪𑼢𑼮𑼶𑼦𑼯𑽂𑼫𑼥𑼴𑼯𑼨𑽂𑼣𑼃𑽃 𑼰𑼒𑼴𑼬𑼾 𑼰𑼞𑼴𑼫𑼡𑼥𑼥𑼶𑼔𑽂𑼬𑼲𑼠𑼴𑼩𑼶𑼙𑽂𑼛𑼙𑽂𑼛𑼴𑼥𑼴𑼮𑼴𑼦𑽂𑼡𑼶𑼯𑼨𑽂𑼣𑼃𑽃 𑼱𑼒𑼴𑼬𑼾 𑼱𑼂𑼮𑼙𑽂𑼛𑼙𑽂𑼛𑼴𑼥𑼴𑼩𑼶𑼱𑼁𑼨𑼾𑼴𑼤𑼥𑼯𑼨𑽂𑼣𑼃𑽃 𑼲𑼒𑼴𑼬𑼾 𑼲𑼡𑼒𑽂𑼭𑼾𑼯𑼮𑼶𑼬𑼴𑼔𑼯𑼨𑽂𑼣𑼃𑽃 𑼒𑽂𑼰𑼒𑼴𑼬𑼾 𑼦𑼬𑼶𑼒𑼷𑼂𑼡𑽂𑼫𑼪𑼴𑼥𑼾 𑼒𑽂𑼰𑼠𑼦𑼂𑼫𑼥𑽂𑼡𑼴𑼩𑼶𑼭𑼴𑼦𑽂𑼫𑼱𑼂𑼮𑼤𑼂𑼪𑼯𑼨𑽂𑼣𑼾𑼴 𑼥𑼶𑼯𑽂𑼗𑼬𑼡𑼶 𑼱𑽂𑼪𑽄

𑼆𑼡𑼶 𑼲𑼶 𑼩𑼶𑼒𑽂𑼰𑼮𑼱𑽂𑼡𑼾𑼰𑼴𑼁 𑼣𑼴𑼬𑼒𑼴𑼠𑼴𑼁 𑼪𑼴𑼡𑼺𑼒𑼴𑼁 𑼮𑼴𑼗𑼫𑼡𑼴𑼁 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼴𑼥𑼸𑼩𑼴𑼮𑼾𑼥𑼿𑼮 𑼦𑽂𑼬𑼪𑼸𑼓𑼴𑼥𑽂𑼫𑼱𑼁𑼓𑽂𑼫𑼾𑼫𑼴𑼥𑼶 𑼤𑼂𑼪𑼪𑼸𑼓𑼯𑼡𑼱𑼲𑼱𑽂𑼬𑼴𑼠𑼶 𑼥𑼶𑼯𑽂𑼗𑼬𑼥𑽂𑼡𑼶 𑼱𑽂𑼪𑽄

𑼡𑼣𑼴𑼥𑼸𑼦𑼹𑼂𑼮𑼾𑼠 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼾𑼥 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼱𑽂𑼢𑼶𑼡𑼾𑼥 𑼣𑽂𑼮𑼴𑼡𑽂𑼬𑼶𑼁𑼯𑼣𑽂𑼣𑼴𑼬𑼒𑼱𑼲𑼱𑽂𑼬𑼴𑼠𑼶 𑼦𑼬𑼶𑼦𑼵𑼗𑼶𑼡𑼴𑼥𑽂𑼫𑼩𑼹𑼮𑼥𑽁𑽃 𑼄𑼥𑼸𑼡𑽂𑼡𑼬𑼴𑼫𑼴𑼁 𑼱𑼪𑽂𑼫𑼒𑽂𑼱𑼁𑼨𑼾𑼴𑼤𑼿𑼴 𑼗𑼶𑼡𑽂𑼡𑼴𑼥𑽂𑼫𑼸𑼡𑽂𑼦𑼵𑼣𑼶𑼡𑼴𑼥𑼶 𑼣𑽂𑼮𑼴𑼡𑽂𑼬𑼶𑼁𑼯𑼣𑽂𑼣𑼴𑼬𑼶𑼒𑼴𑼱𑼲𑼱𑽂𑼬𑼴𑼠𑼶𑽃 𑼄𑼫𑼁 𑼲𑼾𑼡𑼸𑼬𑼫𑼁 𑼦𑽂𑼬𑼡𑽂𑼫𑼫𑼾𑼴 𑼫𑼗𑽂𑼘𑼶𑼒𑽂𑼰𑼶𑼡𑼾𑼴𑼦𑼶 𑼨𑼾𑼴𑼤𑼶𑼱𑼡𑽂𑼡𑽂𑼮𑼾𑼴 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼪𑼸𑼦𑼵𑼔𑼗𑽂𑼘𑼡𑼶 𑼱𑽂𑼪𑽄

𑽄 𑼆𑼡𑼶 𑼯𑽂𑼬𑼷𑼭𑼭𑼶𑼡𑼮𑼶𑼱𑽂𑼡𑼬𑼾 𑼭𑼶𑼦𑼶𑼯𑼴𑼭𑼴𑼱𑼁𑼣𑼂𑼯𑼥𑼦𑼬𑼶𑼮𑼂𑼡𑼾𑼴 𑼥𑼴𑼪 𑼣𑼯𑼪𑼾𑼴𑼤𑽂𑼫𑼴𑼫𑼃𑽄
